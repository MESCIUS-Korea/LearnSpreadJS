시트-레벨 바인딩 | Sheet Level Binding

1. 스프레드 시트는 데이터 소스를 시트에 바인딩하는 것을 지원합니다. 바인딩은 시트 수준 바인딩 및 셀 수준 바인딩 일 수 있습니다.

2. setDataSource 및 getDataSource 메소드를 사용하여 데이터 소스를 설정하고 가져올 수 있습니다. setDataSource를 사용하기 전에 autoGenerateColumns 메서드를 사용하여 데이터를 바인딩하는 동안 열을 자동으로 생성할지 여부를 제어하십시오. 예 : 

var customers = [
     { ID:0, Name:'A', Info1:'Info0' },  
     { ID:1, Name:'B', Info1:'Info1' },  
     { ID:2, Name:'C', Info1:'Info2' },  
 ];
 sheet.autoGenerateColumns = true;
 sheet.setDataSource(customers);

또한 getDataItem 메소드를 사용하여 지정된 행의 데이터 항목을 가져올 수 있습니다.

3. 다음 예제와 같이 지정된 데이터 필드를 사용하여 열을 바인딩 할 수 있습니다. 

var  datasource = [
 {name: 'Alice', age: 27, birthday: '1985/08/31', position: 'PM' }
];
// 하나씩 bindColumn 
var nameColInfo = {name: 'name', dislayName: 'Display Name', size: 70 };
var ageColInfo = {name: 'age', dislayName: 'Age', size: 40, resizable: false };
var birthdayColInfo = {name: 'birthday', dislayName: 'Birthday', formatter: 'd/M/yy', size: 120 };
var positionColInfo = {name: 'position', dislayName: 'Position', size: 50, visible: false };
sheet.autoGenerateColumns = false;
sheet.setDataSource(datasource);
sheet.bindColumn(0, nameColInfo);
sheet.bindColumn(1, birthdayColInfo);
sheet.bindColumn(2, ageColInfo);
sheet.bindColumn(3, positionColInfo);

// 또느 bindColumns를 사용하여 모든 사용자 정의 컬럼을 바인드하십시오.
var colInfos = [
  { name : 'position', displayName: 'Position', size: 50, visible: false },
  { name : 'name', displayName: 'Display Name', size: 70 },
  { name : 'birthday', dislayName: 'Birthday', formatter: 'd/M/yy', size: 120 },
  { name: 'age', dislayName: 'Age', size: 40, resizable: false },
];
sheet.autoGenerateColumns = false;
sheet.setDataSource(datasource);
sheet.bindColumns(colInfos);
